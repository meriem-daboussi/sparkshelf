#### `projects` **[ENHANCED FOR ADMIN MANAGEMENT]**
```sql
CREATE TABLE projects (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  
  -- Basic Info
  title TEXT NOT NULL,
  description TEXT NOT NULL,
  thumbnail_url TEXT NOT NULL,
  detailed_description TEXT, -- Extended content
  
  -- Classification
  difficulty TEXT CHECK (difficulty IN ('beginner', 'intermediate', 'advanced')) NOT NULL,
  estimated_cost_tnd INTEGER NOT NULL,
  time_commitment TEXT CHECK (time_commitment IN ('short', 'medium', 'long')) NOT NULL,
  domain TEXT[] NOT NULL, -- Array: ['mechanical', 'electrical', 'software', 'hybrid', 'aero']
  tags TEXT[], -- ['arduino', 'iot', 'automation']
  
  -- Project Content
  components JSONB NOT NULL, 
  /* Structure:
  [
    {
      "name": "Arduino Nano",
      "quantity": 1,
      "avg_price_tnd": 25,
      "local_suppliers": ["CoThings", "Hobbystart"],
      "notes": "Optional notes"
    }
  ]
  */
  
  steps JSONB NOT NULL,
  /* Structure:
  [
    {
      "step_number": 1,
      "title": "Gather Components",
      "description": "Detailed instructions...",
      "image_url": "optional",
      "estimated_time_minutes": 15
    }
  ]
  */
  
  learning_resources JSONB,
  /* Structure:
  [
    {
      "type": "video" | "article" | "documentation",
      "title": "Arduino Basics",
      "url": "https://...",
      "description": "Optional"
    }
  ]
  */
  
  -- Metadata
  views_count INTEGER DEFAULT 0,
  saves_count INTEGER DEFAULT 0,
  difficulty_score NUMERIC(3,2), -- 1.00 to 5.00 for AI sorting
  
  -- Admin controls
  is_published BOOLEAN DEFAULT FALSE, -- Only published projects appear in feed
  is_featured BOOLEAN DEFAULT FALSE, -- Featured projects appear first
  created_by UUID REFERENCES users(id),
  last_modified_by UUID REFERENCES users(id),
  
  -- SEO
  slug TEXT UNIQUE, -- URL-friendly identifier
  
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Indexes for performance
CREATE INDEX idx_projects_published ON projects(is_published, is_featured, created_at DESC);
CREATE INDEX idx_projects_difficulty ON projects(difficulty) WHERE is_published = TRUE;
CREATE INDEX idx_projects_cost ON projects(estimated_cost_tnd) WHERE is_published = TRUE;
CREATE INDEX idx_projects_domain ON projects USING GIN(domain) WHERE is_published = TRUE;
CREATE INDEX idx_projects_tags ON projects USING GIN(tags) WHERE is_published = TRUE;
CREATE INDEX idx_projects_slug ON projects(slug);

-- Auto-generate slug from title
CREATE OR REPLACE FUNCTION generate_slug()
RETURNS TRIGGER AS $$
BEGIN
  IF NEW.slug IS NULL OR NEW.slug = '' THEN
    NEW.slug := lower(regexp_replace(NEW.title, '[^a-zA-Z0-9]+', '-', 'g'));
    NEW.slug := trim(both '-' from NEW.slug);
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER set_project_slug
  BEFORE INSERT OR UPDATE ON projects
  FOR EACH ROW
  EXECUTE FUNCTION generate_slug();

-- Auto-update timestamp
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = NOW();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER set_updated_at
  BEFORE UPDATE ON projects
  FOR EACH ROW
  EXECUTE FUNCTION update_updated_at();
```

#### `project_revisions` **[NEW - VERSION CONTROL]**
```sql
CREATE TABLE project_revisions (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  project_id UUID REFERENCES projects(id) ON DELETE CASCADE,
  
  -- Snapshot of project data
  revision_data JSONB NOT NULL, -- Complete project JSON
  
  -- Revision metadata
  revision_number INTEGER NOT NULL,
  changed_by UUID REFERENCES users(id),
  change_summary TEXT, -- What was changed
  
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_project_revisions_project ON project_revisions(project_id, revision_number DESC);

-- Auto-increment revision number
CREATE OR REPLACE FUNCTION increment_revision_number()
RETURNS TRIGGER AS $$
BEGIN
  SELECT COALESCE(MAX(revision_number), 0) + 1
  INTO NEW.revision_number
  FROM project_revisions
  WHERE project_id = NEW.project_id;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER set_revision_number
  BEFORE INSERT ON project_revisions
  FOR EACH ROW
  EXECUTE FUNCTION increment_revision_number();
```

### Row Level Security (RLS) Policies **[CRITICAL]**

```sql
-- Enable RLS on all tables
ALTER TABLE users ENABLE ROW LEVEL SECURITY;
ALTER TABLE projects ENABLE ROW LEVEL SECURITY;
ALTER TABLE project_revisions ENABLE ROW LEVEL SECURITY;
ALTER TABLE project_import_logs ENABLE ROW LEVEL SECURITY;

-- USERS TABLE POLICIES
CREATE POLICY "Users can view own profile"
  ON users FOR SELECT
  USING (auth.uid() = id);

CREATE POLICY "Users can update own profile"
  ON users FOR UPDATE
  USING (auth.uid() = id);

-- PROJECTS TABLE POLICIES

-- Public read access to published projects
CREATE POLICY "Anyone can view published projects"
  ON projects FOR SELECT
  USING (is_published = TRUE);

-- Admins can view all projects
CREATE POLICY "Admins can view all projects"
  ON projects FOR SELECT
  USING (
    EXISTS (
      SELECT 1 FROM users
      WHERE users.id = auth.uid()
      AND users.role IN ('admin', 'super_admin')
    )
  );

-- Admins can insert projects
CREATE POLICY "Admins can insert projects"
  ON projects FOR INSERT
  WITH CHECK (
    EXISTS (
      SELECT 1 FROM users
      WHERE users.id = auth.uid()
      AND users.role IN ('admin', 'super_admin')
    )
  );

-- Admins can update projects
CREATE POLICY "Admins can update projects"
  ON projects FOR UPDATE
  USING (
    EXISTS (
      SELECT 1 FROM users
      WHERE users.id = auth.uid()
      AND users.role IN ('admin', 'super_admin')
    )
  );

-- Super admins can delete projects
CREATE POLICY "Super admins can delete projects"
  ON projects FOR DELETE
  USING (
    EXISTS (
      SELECT 1 FROM users
      WHERE users.id = auth.uid()
      AND users.role = 'super_admin'
    )
  );

-- PROJECT REVISIONS POLICIES
CREATE POLICY "Admins can view all revisions"
  ON project_revisions FOR SELECT
  USING (
    EXISTS (
      SELECT 1 FROM users
      WHERE users.id = auth.uid()
      AND users.role IN ('admin', 'super_admin')
    )
  );

CREATE POLICY "Admins can insert revisions"
  ON project_revisions FOR INSERT
  WITH CHECK (
    EXISTS (
      SELECT 1 FROM users
      WHERE users.id = auth.uid()
      AND users.role IN ('admin', 'super_admin')
    )
  );

-- IMPORT LOGS POLICIES
CREATE POLICY "Admins can view import logs"
  ON project_import_logs FOR SELECT
  USING (
    EXISTS (
      SELECT 1 FROM users
      WHERE users.id = auth.uid()
      AND users.role IN ('admin', 'super_admin')
    )
  );
```

```typescript
// ============================================

    saves: number,
    completion_rate: number
  }

// Create new project
POST /api/admin/projects
Body: {
  title: string,
  description: string,
  detailed_description?: string,
  thumbnail_url: string,
  difficulty: 'beginner' | 'intermediate' | 'advanced',
  estimated_cost_tnd: number,
  time_commitment: 'short' | 'medium' | 'long',
  domain: string[],
  tags: string[],
  components: Component[],
  steps: Step[],
  learning_resources?: Resource[],
  is_published: boolean,
  is_featured: boolean
}
Response: {
  project: Project,
  success: boolean
}

// Update existing project
PATCH /api/admin/projects/:id
Body: Partial<Project> & {
  change_summary: string // Required for revision tracking
}
Response: {
  project: Project,
  revision: ProjectRevision,
  success: boolean
}

// Delete project (soft delete - mark as unpublished)
DELETE /api/admin/projects/:id
Query params: {
  permanent?: boolean // Only super_admin can permanently delete
}
Response: {
  success: boolean
}

// Bulk publish/unpublish
PATCH /api/admin/projects/bulk-status
Body: {
  project_ids: string[],
  is_published: boolean
}
Response: {
  updated_count: number,
  success: boolean
}

// Duplicate project (for quick variations)
POST /api/admin/projects/:id/duplicate
Body: {
  title_suffix?: string // e.g., " - Copy"
}
Response: {
  project: Project,
  success: boolean
}

// BULK IMPORT/EXPORT

}
Response: {

  failed: number,

    project_title: string,
    error: string
  }>

Query params: {

Response: File download

GET /api/admin/import-logs
Response: {
  logs: ImportLog[]
}
```

```typescript
// src/services/admin.service.ts

import { supabase } from './supabase';
import type { 
  Project, 
  CreateProjectDTO, 
  UpdateProjectDTO, 
  BulkImportDTO,
  ImportLog 
} from '../types/project.types';

export class AdminService {
  // ============================================
  // PROJECT CRUD
  // ============================================
  
  async getAllProjects(params: {
    limit?: number;
    offset?: number;
    is_published?: boolean;
    difficulty?: string;
    search?: string;
  }) {
    let query = supabase
      .from('projects')
      .select('*', { count: 'exact' })
      .order('created_at', { ascending: false });

    if (params.is_published !== undefined) {
      query = query.eq('is_published', params.is_published);
    }

    if (params.difficulty) {
      query = query.eq('difficulty', params.difficulty);
    }

    if (params.search) {
      query = query.or(`title.ilike.%${params.search}%,description.ilike.%${params.search}%`);
    }

    if (params.limit) {
      query = query.range(params.offset || 0, (params.offset || 0) + params.limit - 1);
    }

    const { data, error, count } = await query;

    if (error) throw error;

    return {
      projects: data as Project[],
      total: count || 0,
      hasMore: (count || 0) > (params.offset || 0) + (params.limit || 10)
    };
  }

  async getProjectById(id: string) {
    const { data, error } = await supabase
      .from('projects')
      .select(`
        *,
        revisions:project_revisions(*)
      `)
      .eq('id', id)
      .single();

    if (error) throw error;
    return data;
  }

  async createProject(dto: CreateProjectDTO) {
    const { data: user } = await supabase.auth.getUser();
    
    const { data, error } = await supabase
      .from('projects')
      .insert({
        ...dto,
        created_by: user.user?.id,
        last_modified_by: user.user?.id
      })
      .select()
      .single();

    if (error) throw error;

    // Create initial revision
    await this.createRevision(data.id, data, 'Initial creation');

    return data as Project;
  }

  async updateProject(id: string, dto: UpdateProjectDTO) {
    const { data: user } = await supabase.auth.getUser();
    
    const { change_summary, ...updateData } = dto;

    const { data, error } = await supabase
      .from('projects')
      .update({
        ...updateData,
        last_modified_by: user.user?.id
      })
      .eq('id', id)
      .select()
      .single();

    if (error) throw error;

    // Create revision
    await this.createRevision(id, data, change_summary);

    return data as Project;
  }

  async deleteProject(id: string, permanent: boolean = false) {
    if (permanent) {
      // Permanent delete (super admin only)
      const { error } = await supabase
        .from('projects')
        .delete()
        .eq('id', id);

      if (error) throw error;
    } else {
      // Soft delete - just unpublish
      await this.updateProject(id, { 
        is_published: false,
        change_summary: 'Project unpublished'
      });
    }

    return { success: true };
  }

  async duplicateProject(id: string, titleSuffix: string = ' - Copy') {
    const original = await this.getProjectById(id);
    
    const { id: _, slug: __, created_at: ___, updated_at: ____, ...projectData } = original;

    const newProject = await this.createProject({
      ...projectData,
      title: `${original.title}${titleSuffix}`,
      is_published: false // Duplicates start as drafts
    });

    return newProject;
  }

  // ============================================
  // BULK OPERATIONS
  // ============================================

  async bulkUpdateStatus(dto: BulkStatusUpdateDTO) {
    const { data, error } = await supabase
      .from('projects')
      .update({ is_published: dto.is_published })
      .in('id', dto.project_ids)
      .select();

    if (error) throw error;

    return {
      updated_count: data.length,
      success: true
    };
  }

  async bulkImport(dto: BulkImportDTO): Promise<ImportLog> {
    const { data: user } = await supabase.auth.getUser();
    const errors: Array<{ project_title: string; error: string }> = [];
    let successful = 0;

    for (const projectSeed of dto.projects) {
      try {
        // Check if project exists (by title or slug)
        if (dto.overwrite_existing) {
          const { data: existing } = await supabase
            .from('projects')
            .select('id')
            .eq('title', projectSeed.title)
            .single();

          if (existing) {
            // Update existing
            await this.updateProject(existing.id, {
              ...projectSeed,
              is_published: dto.auto_publish,
              change_summary: 'Bulk import update'
            });
            successful++;
            continue;
          }
        }

        // Create new project
        await this.createProject({
          ...projectSeed,
          is_published: dto.auto_publish
        });
        successful++;
      } catch (error) {
        errors.push({
          project_title: projectSeed.title,
          error: error instanceof Error ? error.message : 'Unknown error'
        });
      }
    }

    // Log the import
    const { data: log } = await supabase
      .from('project_import_logs')
      .insert({
        imported_by: user.user?.id,
        file_name: 'bulk_import.json',
        total_projects: dto.projects.length,
        successful_imports: successful,
        failed_imports: errors.length,
        error_log: errors
      })
      .select()
      .single();

    return log as ImportLog;
  }

  async exportProjects(includeUnpublished: boolean = false) {
    let query = supabase.from('projects').select('*');

    if (!includeUnpublished) {
      query = query.eq('is_published', true);
    }

    const { data, error } = await query;

    if (error) throw error;

    // Return as JSON string for download
    return JSON.stringify(data, null, 2);
  }

  // ============================================
  // REVISION MANAGEMENT
  // ============================================

  private async createRevision(projectId: string, projectData: any, changeSummary: string) {
    const { data: user } = await supabase.auth.getUser();

    const { error } = await supabase
      .from('project_revisions')
      .insert({
        project_id: projectId,
        revision_data: projectData,
        changed_by: user.user?.id,
        change_summary: changeSummary
      });

    if (error) throw error;
  }

  async getRevisions(projectId: string) {
    const { data, error } = await supabase
      .from('project_revisions')
      .select('*')
      .eq('project_id', projectId)
      .order('revision_number', { ascending: false });

    if (error) throw error;
    return data;
  }

  async restoreRevision(projectId: string, revisionId: string) {
    // Get the revision
    const { data: revision, error: revError } = await supabase
      .from('project_revisions')
      .select('revision_data')
      .eq('id', revisionId)
      .single();

    if (revError) throw revError;

    // Update project with revision data
    const { id, created_at, updated_at, ...restoreData } = revision.revision_data;

    return this.updateProject(projectId, {
      ...restoreData,
      change_summary: `Restored to revision #${revision.revision_data.revision_number}`
    });
  }

  // ============================================
  // STATISTICS
  // ============================================

  async getProjectStats(projectId: string) {
    // Get views, saves, completion rate
    const { data: project } = await supabase
      .from('projects')
      .select('views_count, saves_count')
      .eq('id', projectId)
      .single();

    const { count: completions } = await supabase
      .from('user_projects')
      .select('*', { count: 'exact', head: true })
      .eq('project_id', projectId)
      .eq('status', 'completed');

    const { count: saves } = await supabase
      .from('user_projects')
      .select('*', { count: 'exact', head: true })
      .eq('project_id', projectId);

    return {
      views: project?.views_count || 0,
      saves: project?.saves_count || 0,
      completions: completions || 0,
      completion_rate: saves ? ((completions || 0) / saves) * 100 : 0
    };
  }
}

export const adminService = new AdminService();
```

```typescript
// src/components/Admin/ProjectEditor/ComponentsEditor.tsx

import React, { useState } from 'react';
import type { Component } from '@/types/project.types';

interface ComponentsEditorProps {
  components: Component[];
  onChange: (components: Component[]) => void;
}

export const ComponentsEditor: React.FC<ComponentsEditorProps> = ({ 
  components, 
  onChange 
}) => {
  const [editingIndex, setEditingIndex] = useState<number | null>(null);

  const addComponent = () => {
    onChange([
      ...components,
      {
        name: '',
        quantity: 1,
        avg_price_tnd: 0,
        local_suppliers: [],
        notes: ''
      }
    ]);
    setEditingIndex(components.length);
  };

  const updateComponent = (index: number, field: keyof Component, value: any) => {
    const updated = [...components];
    updated[index] = { ...updated[index], [field]: value };
    onChange(updated);
  };

  const deleteComponent = (index: number) => {
    onChange(components.filter((_, i) => i !== index));
  };

  const moveComponent = (index: number, direction: 'up' | 'down') => {
    const newIndex = direction === 'up' ? index - 1 : index + 1;
    if (newIndex < 0 || newIndex >= components.length) return;

    const updated = [...components];
    [updated[index], updated[newIndex]] = [updated[newIndex], updated[index]];
    onChange(updated);
  };

  return (
    <div className="space-y-4">
      <div className="flex justify-between items-center">
        <h3 className="text-lg font-mono text-primary">Components List</h3>
        <button
```

```typescript
import React, { useState } from 'react';
import { supabase } from '@/services/supabase';

interface MediaUploaderProps {
  label: string;
  currentUrl?: string;
  onUpload: (url: string) => void;
  accept?: string;
}

export const MediaUploader: React.FC<MediaUploaderProps> = ({
  label,
  currentUrl,
  onUpload,
  accept = 'image/*'
}) => {
  const [uploading, setUploading] = useState(false);
  const [progress, setProgress] = useState(0);

  const handleUpload = async (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (!file) return;

    setUploading(true);
    setProgress(0);

    try {
      // Generate unique filename
      const fileExt = file.name.split('.').pop();
      const fileName = `${Math.random().toString(36).substring(2)}-${Date.now()}.${fileExt}`;
      const filePath = `projects/${fileName}`;

      // Upload to Supabase Storage
      const { data, error } = await supabase.storage
        .from('media')
        .upload(filePath, file, {
          cacheControl: '3600',
          upsert: false
        });

      if (error) throw error;

      // Get public URL
      const { data: { publicUrl } } = supabase.storage
        .from('media')
        .getPublicUrl(filePath);

      onUpload(publicUrl);
      setProgress(100);
    } catch (error) {
      console.error('Upload failed:', error);
      alert('Upload failed. Please try again.');
    } finally {
      setUploading(false);
    }
  };

  return (
    <div>
      <label className="block text-sm font-mono text-gray-300 mb-2">
        {label}
      </label>

      {currentUrl && (
        <div className="mb-3 relative">
          <img 
            src={currentUrl} 
            alt="Preview" 
            className="w-full max-w-md h-48 object-cover rounded border border-gray-700"

            type="button"
```

## 1ï¸âƒ£1ï¸âƒ£ Admin Hook Implementation

```typescript
// src/hooks/useAdminPermissions.ts

import { useEffect, useState } from 'react';
import { supabase } from '@/services/supabase';

export const useAdminPermissions = () => {
  const [isAdmin, setIsAdmin] = useState(false);
  const [isSuperAdmin, setIsSuperAdmin] = useState(false);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    checkAdminStatus();
  }, []);

  const checkAdminStatus = async () => {
    try {
      const { data: { user } } = await supabase.auth.getUser();
      
      if (!user) {
        setIsAdmin(false);
        setIsSuperAdmin(false);
        setLoading(false);
        return;
      }

      const { data, error } = await supabase
        .from('users')
        .select('role')
        .eq('id', user.id)
        .single();

      if (error) throw error;

      setIsAdmin(data.role === 'admin' || data.role === 'super_admin');
      setIsSuperAdmin(data.role === 'super_admin');
    } catch (error) {
      console.error('Error checking admin status:', error);
      setIsAdmin(false);
      setIsSuperAdmin(false);
    } finally {
      setLoading(false);
    }
  };

  return { isAdmin, isSuperAdmin, loading };
};
```
